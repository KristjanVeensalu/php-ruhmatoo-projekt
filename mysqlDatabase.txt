CREATE TABLE userSample(
	id NOT NULL AUTO_INCREMENT,
	username VARCHAR(50) NOT NULL PRIMARY KEY,
	password VARCHAR(128),
	created timestamp DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE trackFood(
	username VARCHAR(50),
	food VARCHAR(50),
	calories INT,
	FOREIGN KEY(username) REFERENCES userSample(username)
);

CREATE TABLE trackAbs(
	username VARCHAR(50),
	reps INT,
);

CREATE TABLE trackBack(
	username VARCHAR(50),
	reps INT,
);

CREATE TABLE trackChest(
	username VARCHAR(50),
	reps INT,
);

CREATE TABLE trackCore(
	username VARCHAR(50),
	reps INT,
);

CREATE TABLE trackLegs(
	username VARCHAR(50),
	reps INT,
);

CREATE TABLE trackShoulders(
	username VARCHAR(50),
	reps INT,
);
